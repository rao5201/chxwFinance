创建你的第一个插件

typescript 
import { Context, Plugin } from '@chxw/core';
import { StockAPI } from '@chxw/finance';

export default class StockPlugin implements Plugin {
  private stockAPI: StockAPI;

  constructor() {
    this.stockAPI = new StockAPI();
  }

  // 插件名称
  name = 'stock-query';

  // 插件配置
  config = {
    defaultMarket: 'US',
    enableRealtime: true
  };

  // 消息处理
  async onMessage(ctx: Context) {
    const { message } = ctx;
    
    // 匹配股票查询命令
    if (message.content.startsWith('/stock')) {
      const symbol = message.content.split(' ')[1];
      
      // 获取股票数据
      const data = await this.stockAPI.getQuote(symbol);
      
      // 回复用户
      await ctx.reply(`📈 ${symbol}: $${data.price} (${data.change}%)`);
    }
  }
}

TypeScript 支持

完全基于 TypeScript 开发，拥有顶级的类型支持。丰富的代码提示让你在编写代码的时候甚至无需查看文档。

类型定义示例

typescript 
// 上下文类型定义
interface Context {
  // 消息对象
  message: Message;
  
  // 发送者信息
  sender: User;
  
  // 会话信息
  session: Session;
  
  // 回复消息
  reply(content: string): Promise<void>;
  
  // 发送图片
  sendImage(url: string): Promise<void>;
  
  // 获取数据库连接
  database: Database;
}

// 插件接口
interface Plugin {
  name: string;
  config?: Record<string, any>;
  onMessage?(ctx: Context): Promise<void> | void;
  onCommand?(cmd: string, ctx: Context): Promise<void> | void;
}

配置热重载

typescript 
import { defineConfig } from '@chxw/core';

export default defineConfig({
  // 启用热重载
  hotReload: {
    enabled: true,
    // 监听的文件模式
    patterns: ['src/**/*.ts', 'plugins/**/*.ts'],
    // 忽略的文件
    ignore: ['node_modules', '*.test.ts'],
    // 延迟重载（毫秒）
    delay: 300
  },
  
  // 开发服务器配置
  devServer: {
    port: 3000,
    // 启用沙盒界面
    sandbox: true
  }
});
